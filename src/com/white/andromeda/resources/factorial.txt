//Программа по вычислению факториала

x = 10;             //Переменная, факториал которой будет найден
fact = 1;           //Переменная для факториала
while (x > 0){      //Цикл для факториала
    fact *= x;
    if (x > 3){
        if (fact < 1000)
            print x;
    }else
        print fact;
    x -= 1;
}
print fact;
